namespace Roulette;

use Roulette\Iterator;
use Roulette\Collection;

/**
 * Provide a view standart function. Top level parent class for Roulette Classes.
 * 
 * @package \Roulette
 * @author Eko Dedy Purnomo (eko.dedy.purnomo@gmail.com)
 * @since Version 0.1.0
 */
class Base
{
    /**
     * Function to send new data
     * 
     * @param [type] $config
     * @return [type]        
     */
    static function create($config = null)
    {
        return new static($config);
    }

    /**
     * Determines information about the current platform the application is running on
     * 
     * @param  object  $object
     * @return object
     */
    static function is($object = null)
    {
        return is_object($object) and (is_a($object, static::class) or is_subclass_of($object, static::class));
    }

    /**
     * [isNot description]
     * 
     * @param  object $object
     * @return object
     */
    static function isNot($object = null)
    {
        return !static::is($object);
    }

    /**
     * function to process layout
     * 
     * @param  object $object
     * @return object
     */
    static function from($object = null)
    {
        if( !static::is($object) ){
            $object = new static($object);
        }
        return $object;
    }

}
