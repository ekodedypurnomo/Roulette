namespace Roulette;

use Roulette\Base;
use Roulette\Iterator;
use Roulette\Proxy;

/**
 * Helper can assist in using this roulette there is a shortcut, among others. 
 * call(), each(), apply(), applyIf()
 * Where without having to call its parent class
 *
 * @package Roulette
 * @since Version 0.1.0
 * @author Eko Dedy Purnomo <eko.dedy.purnomo@gmail.com>
 */
class Helper extends Base
{
    /**
     * Shortcut to Proxy::tunnel 
     * 
     * @return Array
     */
    static function tunnel()
    {
        return call_user_func_array(array(Proxy::class, 'tunnel'), func_get_args());
    }

    /**
     * Call a function safely.
     * 
     * @param  callable|null $function
     * @param  array|null    $params  
     * @return mixed
     */
	static function callback(callable $function = null, array $params = null)
	{
        if (is_callable($function))
        {
            if (!is_array($params)) $params = array($params);

            return call_user_func_array($function, $params);
        }
    }

    /**
     * Shortcut to callback
     *
     * @return mixed 
     */
    static function call()
    {
        return forward_static_call_array(array(static::class, 'callback'), func_get_args());
    }

    /**
     * Shortcut to each Iterator
     * 
     * @return array
     */
    static function each()
    {
    	return call_user_func_array(array(Iterator::class, 'each'), func_get_args());
    }

    /**
     * Shortcut to apply Iterator
     * 
     * @return array
     */
    static function apply()
    {
    	return call_user_func_array(array(Iterator::class, 'apply'), func_get_args());
    }

    /**
     * Shortcut to applyIf Iterator
     * 
     * @return array
     */
    static function applyIf()
    {
    	return call_user_func_array(array(Iterator::class, 'applyIf'), func_get_args());
    }

    /**
     * Create an alias of function and return it as a function.
     * 
     * @param  array $function
     * @return array
     */
    static function createAlias($function = null)
    {
        return function() use ($function){
            return call_user_func_array( $function, func_get_args() );
        };
    }

    /**
     * create an alias of function with a new function name
     * 
     * @param  String $original
     * @param  String $target  
     * @return Array
     */
    static function alias($original, $target) {
        eval("function $target() { \$args = func_get_args(); return call_user_func_array('$original', \$args); }");
    }
}
