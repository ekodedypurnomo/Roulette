    /**
     * @ignore
     */
    static function initialize(array $initConfig = null)
    {
        $class = static::class;
        
        if(is_object($initConfig)) $initConfig = (object) $initConfig;
        if(empty($initConfig) || !is_array($initConfig)) $initConfig = array();

        if(is_object(static::$prototype)) static::$prototype = (array) static::$prototype;
        if(empty(static::$prototype)) static::$prototype = array();
        
        $configs = Collection::from(array_merge(static::$prototype, $initConfig));

        # initializing table
        if($configs->has('table'))
        {
            static::setTable($configs->get('table'));
        }        

        # initializing primary
        if($configs->has('primary'))
        {
            static::setPrimary($configs->get('primary'));
        }

        # initializing fields
        $fields = static::getFields()->reset();

        I::each($configs->get('fields'), function($f, $i, $all) use($class, $fields)
        {
            $f = Field::from($f);
            $fields->set($f->name, $f);
        });

        # initializing associations
        $associations = static::getAssociations()->reset();

        I::each($configs->get('associations'), function($a, $i, $all) use($class, $associations)
        {
            $a = Association::from($a);
            $associations->set($a->getName(), $a);
        });

        # returning back the class
        return static::class;
    }

    /**
     * @ignore
     */
    static function prototype()
    {
        if(func_num_args() == 0)
        {
            if(Collection::isNot(static::$prototype))
            {
                static::$prototype = Collection::create();
            }

            return static::$prototype;
        }
        else
        {
            $prototypeConfig = func_get_arg(0);
            return static::initialize($prototypeConfig);
        }
    }

    /**
     * Get the attribute for every fields in the model
     * 
     * @param  string $attribute 
     * @param  string|array $fields     
     * @return array           
     */
    static function getFieldsAttribute($attribute = null, $fields = null)
    {
        if( is_string($fields) )
        {
            $fields = strtolower($fields);

            if( $fields == 'all' )
            {
                $collectedAttribute = array();

                I::each(static::findFields(), function($f) use( &$collectedAttribute, $attribute)
                {
                    $collectedAttribute[$f->name] = property_exists($f, $attribute) ? $f->$attribute : null;
                });

                return $collectedAttribute;
            }
            else if( $fields == 'original')
            {
                $collectedAttribute = array();

                I::each(static::findFields(), function($f) use( &$collectedAttribute, $attribute)
                {
                    if($f->extension != true)
                    {
                        $collectedAttribute[$f->name] = property_exists($f, $attribute) ? $f->$attribute : null;
                    }
                });

                return $collectedAttribute;
            }
        }
        elseif( is_array($fields) )
        {
            $collectedAttribute = array();

            I::each(static::findFields($fields), function($f) use( &$collectedAttribute, $attribute)
            {
                $collectedAttribute[$f->name] = property_exists($f, $attribute) ? $f->$attribute : null;
            });

            return $collectedAttribute;
        }
        else
        {
            $f = static::getFields()->get($fields);
            
            return $f && property_exists($f, $attribute) ? $f->$attribute : null;
        }
    }

    /**
     * Get the attribute from a specified field
     * 
     * @param  string $field     
     * @param  String $attribute 
     * @return string            
     */
    static function getFieldAttribute($field, $attribute)
    {
        if(!is_string($attribute)) return;
        $field = static::getField($field);
        return ($field and property_exists($field, $attribute)) ? $field->$attribute : null;  
    }

    /**
     * find fields from the model
     * @return array
     */
    static function findFields()
    {
        # all fields
        if(func_num_args() == 0)
        {
            return static::getFields()->getAll();
        }


        $arg1 = func_get_arg(1);

        # if arguments is fields
        if(Field::is($arg1))
        {
            if(static::getFields()->contain($arg1)) return $arg1;
        }
        # if function of condition
        elseif(is_callable($arg1))
        {
            return static::getFields()->filter($arg1);        
        }
        # if bulk of fields
        elseif(is_array($arg1))
        {
            $fields = array();
            foreach ($arg1 as $i => $f)
            {
                # if field instance
                if( Field::is($f) and static::getFields()->contain($f))
                {
                    $fields[] = $f;
                }
                # if field name string
                elseif(is_string($f) and $found = static::getFields()->get($f))
                {
                    $fields[] = $found;
                }
            }
        }
    }

    /**
     * Find fields of the model that specified by the condition of the parameter
     * @param  array $condition 
     * @return boolean            
     */
    static function findFieldsIf($condition)
    {
        if(is_array($condition))
        {
            return static::getFields()->filter(function($field) use($condition)
            {
                $valid = true;

                foreach ($condition as $fKey => $fValue) {
                    if(! (property_exists($field, $fKey) and $field->$fKey === $fValue) )
                    {
                        $valid = false;
                        break;
                    }
                }

                return $valid;
            });
        }
    }

    /**
     * [mapToSource description]
     * @param  string|array $data 
     * @return [type]       
     */
    static function mapToSource($data = null)
    {
        $fieldsSource = static::getFieldsSource();
        $mapped = null;

        if(is_string($data))
        {
            $mapped = Template::compile($data, $fieldsSource);
        }
        if(is_array($data))
        {
            $mapped = array();
            foreach ($data as $key => $value) 
            {
                # compile if key contains source 
                $key = array_key_exists($key, $fieldsSource) ? $fieldsSource[$key] : Template::compile($key)->apply($fieldsSource);
                
                # compile if value contains source
                $value = Template::compile($value)->apply($fieldsSource);
                
                # reorder all in new array
                $mapped[$key] = $value;
            }
        }

        return $mapped;
    }

    /**
     * [genIdCache description]
     * @param  [type] $id [description]
     * @return [type]     [description]
     */
    static function genIdCache($id)
    {
        $format = get_class().static::class.'-'.$id;

        return $format;
    }

    /**
     * [storeCache description]
     * @param  [type] $record [description]
     * @return [type]         [description]
     */
    static function storeCache($record)
    {
        if(static::useCache())
        {
            Cache::store(static::genIdCache($record->getId()), $record);
        }
        return static::class;
    }

    /**
     * [fetchCache description]
     * @param  [type] $id [description]
     * @return [type]     [description]
     */
    static function fetchCache($id)
    {
        if(static::useCache())
        {
            if($_c = Cache::fetch(static::genIdCache($id)))
            {
                return $_c;
            }
        }
    }

    /**
     * [load description]
     * @param  [type] $id [description]
     * @return [type]     [description]
     */
    static function load( $id = null )
    {
        # chaching check before goes deep
        if($_c = static::fetchCache($id))
        {
            return $_c;
        }

        $condition = is_array($id) ? $id : array( static::getPrimary() => $id );

        $operation = Proxy::selectOne(array(
            'table' => static::getTable(),
            'fields' => static::getFieldsSource('original'),
            'condition' => static::mapToSource($condition),
        ));

        if( $operation->getRecord() )
        {
            # use __construct instead of create, we need to include `$original = true` in the object  
            $record = new static((array) $operation->getRecord(), true);
            return $record;
        }
    }

    /**
     * [find description]
     * @param  array   $condition [description]
     * @param  [type]  $order     [description]
     * @param  [type]  $take      [description]
     * @param  integer $start     [description]
     * @param  [type]  $group     [description]
     * @param  [type]  $having    [description]
     * @return [type]             [description]
     */
    static function find($condition = array(), $order = null, $take = null, $start = 0, $group = null, $having = null)
    {
        $operation = Proxy::select(array(
            'table' => static::getTable(),
            'fields' => static::getFieldsSource(),
            'condition' => static::mapToSource($condition),
            'take' => $take,
            'start' => $start,
            'order' => $order,
            'group' => $group,
            'having' => $having
        ));
        
        $records = Collection::from($operation->getRecords());
        $records->each(function($v, $k, $all, $c)
        {
            $r = new static((array)$v, true);
            $c->set($k, $r);
        });

        return $records->getAll();
    }

    /**
     * @ignore
     */
    function __construct(Array $data = null, $original = false)
    {
        # set up all data
        $this->initData($data, $original);

        # then make it alive based on original status
        if($original)
        {
            $this->makeAlive();
        }

        return $this;
    }

    /**
     * Check the record whether it's modified or not
     * @return boolean 
     */
    function isModified()
    {
        return $this->data()->some(function($data)
        {
            return $data->isModified();
        });
    }

    /**
     * Get the modified field from the record
     * @return array
     */
    function getModified()
    {
        $modified = array();
        $this->data()->each(function($data, $fieldName) use(&$modified)
        {
            if($data->isModified()) $modified[] = $fieldName;
        });
        return $modified;
    }

    /**
     * [getErrorMessages description]
     * @param  boolean $grouped [description]
     * @return [type]           [description]
     */
    function getErrorMessages($grouped = false)
    {
        $errorMessages = array();

        $this->data()->each(function($fieldValue, $key) use(&$errorMessages, $grouped)
        {
            if(!$fieldValue->isValid())
            {
                if($grouped)
                {
                    $errorMessages[$key] = $fieldValue->getErrorMessages();
                }
                else
                {
                    $errorMessages = array_merge($errorMessages, $fieldValue->getErrorMessages());
                }
            }
        });

        return $errorMessages;
    }

    /**
     * Check the record whether it's valid or not
     * @param boolean runvalidate 
     * @return boolean 
     */
    function isValid($runValidate = false)
    {
        if($runValidate) $this->validate();
        
        return $this->data()->every(function($fieldValue){
            return $fieldValue->isValid();
        });
    }

    /**
     * Check whether the record is exist in the database or not
     * force recheck will not affect any changes on database into object or vise versa, it will only validate if id is exist.
     * 
     * @param  boolean $recheck 
     * @return boolean           
     */
    function isAlive($recheck = false)
    {
        if ($recheck and $this->hasId()) # check will be available if only record has an id
        {
            # only renew original values from database, but accept user changes
            $this->reload($revert = false);
        }

        return $this->alive;
    }

    /**
     * [makeAlive description]
     * @param  boolean $alive [description]
     * @return [type]         [description]
     */
    protected function makeAlive($alive = true)
    {
        $this->alive = !!$alive;

        if($this->alive)
        {
            static::storeCache($this);
        }

        return $this;
    }

    /**
     * Check the record if it has id
     * @return boolean [description]
     */
    function hasId()
    {
        $id = $this->getId();
        return (is_int($id) || (is_string($id) and !empty($id)));
    }

    /**
     * [initData description]
     * @param  array|null $data     [description]
     * @param  boolean    $original [description]
     * @return [type]               [description]
     */
    protected function initData(array $data = null, $original = false)
    {
        $me = $this;
    
        $data = Collection::from($data);

        $this->getFields()->each(function($field, $i) use($me, $data, $original)
        {
            $value = $data->get($field->name);

            # does no need use _set, affect original first then raw instead
            if( $fieldValue = $me->_get($field->name) )
            {
                if($original)
                {
                    $fieldValue->setOriginal($value);
                    $fieldValue->setValue($fieldValue->getOriginal()); # set raw equal original;
                }
                else
                {
                    $fieldValue->setValue($value);
                }
            }
        });
        
        return $this;
    }

    /**
     * an existing record or new record on the model
     * @return [type] [description]
     */
    protected function data()
    {
        if(Collection::isNot($this->data)) $this->data = new Collection();
        return $this->data;
    }

    /**
     * life cycle of field data
     *
     * @see \Roulette\field\Value FieldValue value life cycle
     * @param [type] $field    [description]
     * @param [type] $value    [description]
     * @param boolean $commit   [description]
     * @param boolean $original [description]
     */
    protected function _set($field, $value = null, $commit = false, $original = false)
    {
        # bulk operation
        if(is_object($field)) $field = (array) $field;
        if (is_array($field))
        {
            foreach ($field as $f => $v)
            {
                $this->_set($f, $v, $commit, $original);
            }
            return $this;
        }

        # single operation
        if( $fieldValue = $this->_get($field) )
        {
            if($original)
            {
                $fieldValue->setOriginal($value, $revert = false);
            }
            else
            {
                $fieldValue->setValue($value, $commit);
            }
        }

        return $this;
    }

    /**
     * Set the value of the record by specified name
     * @param string $field 
     * @param string $value 
     */
    function set($field, $value = null)
    { 
        # use _set instead, for prevent advanced set
        $this->_set($field, $value, $commit = false, $original = false);

        return $this;
    }

    /**
     * @ignore
     */
    protected function _get($field = null)
    {
        if ($foundField = static::getField($field))
        {
            $data = $this->data();

            if (!$data->hasKey($foundField->name)) $data->set($foundField->name, new FieldValue($this, $foundField));

            return $data->get($foundField->name);
        }
    }

    /**
     * get the record from the specified field name
     * @param  array  $field  
     * @param  boolean $render 
     * @return array          
     */
    function get($field = null, $render = true)
    {
        if(is_array($field))
        {
            $data = array();
            foreach ($field as $i => $f)
            {
                $data[$f] = $this->get($f, $render);
            }
            return $data;
        }

        if($fieldValue = $this->_get($field))
        {
            return $render ? $fieldValue->getDisplay() : $fieldValue->getRaw();
        }
    }

    /**
     * Get the record from the model
     * @param  string|boolean  $fields 
     * @param  boolean $render 
     * @return array         
     */
    function getData($fields = null, $render = true)
    {
        if(is_bool($fields))
        {
            $render = $fields;
            $fields = null;
        }

        if (is_string($fields)) {
            $fields = [$fields];
        }

        if (empty($fields)) {
            $fields = static::findFieldsPublic(true);
        }

        if ( !$fields) return; # only accept get data from registered fields

        $data = array();
        foreach ($fields as $key => $fieldName) {

            $data[$fieldName] = $this->get($fieldName, $render);
        }
        return $data;
    }

    /**
     * [clear description]
     * @param  boolean $removeValue [description]
     * @return [type]               [description]
     */
    function clear()
    {
        $this->data()->reset();
        return $this;
    }

    /**
     * [reset description]
     * @return [type] [description]
     */
    function reset()
    {
        return $this->clear();
    }

    /**
     * [revert description]
     * @return [type] [description]
     */
    function revert()
    {
        $me = $this; 
        $this->data()->each(function($fieldValue)
        {
            $fieldValue->revert();
        });
        return $this;
    }

    /**
     * [commit description]
     * @param boolean $makeAlive [description]
     * @return  [description]
     */
    function commit( $makeAlive = false )
    {
        $me = $this; 
        $this->data()->each(function($data){
            $data->commit();
        });
        if($makeAlive) $this->makeAlive();
        return $this;
    }

    /**
     * Refetch the record from DB
     * @param  boolean $revert
     * @param  function $callback
     * @return /Roulette/Model          
     */
    function reload($revert = true, $callback = null)
    {
        if (!$this->hasId()) return $this;

        # if callback as first param
        if(is_callable($revert))
        {
            $callback = $revert;
            $revert = true; # as the default of revert
        }

        # use operation instead of load
        $condition = array(
            static::getPrimary() => $this->getId()
        );
        
        $operation = Proxy::selectOne(array(
            'table' => static::getTable(),
            'fields' => static::getFieldsSource('original'),
            'condition' => static::mapToSource($condition),
        ));

        if($operation->isSuccess())
        {
            # changes alive status by the record existense
            $this->makeAlive(!!$operation->getRecord());

            $rawRecord = (array)$operation->getRecord(); # parse into array for bulk set
            
            # update original data with the new from database
            $this->_set($rawRecord, $_ignoreit = null, $commit = false, $original = true);
            
            # revert if needed
            if($revert)
            {
                $this->revert();
            }
        }

        Helper::callback($callback, array($this));

        return $this;
    }

    /**
     * Check the record is valid or not
     * @param  function $callback 
     * @return /Roulette/Model           
     */
    function validate( $callback = null )
    {
        $valid = $this->data()->every(function($fieldValue, $fieldName) use(&$error)
        {
            $fieldValue->validate();
            return $fieldValue->isValid();
        });

        Helper::callback($callback, array($valid, $this));

        return $this;
    }

    /**
     * [getDataToSave description]
     * @param  string  $operationMode [description]
     * @param  boolean $modifiedOnly  [description]
     * @return [type]                 [description]
     */
    function getDataToSave( $operationMode = 'save', $modifiedOnly = false) 
    {
        $operationMode = strtolower($operationMode);

        $dataToSave = array();
        
        $this->data()->each(function($v, $k) use(&$dataToSave, $operationMode, $modifiedOnly)
        {
            $f = $v->getField();

            # exlude any extension field
            if($f->isExtension())
            {
                return;   
            }

            # exlude any unchanges if modifiedOnly is true
            if($modifiedOnly and !$v->isModified())
            {
                return;
            }

            # fetch by its operationMode
            if( 
                ($operationMode == 'save' and ($f->insertable() or $f->updateable())) or
                ($operationMode == 'insert' and $f->insertable() ) or   
                ($operationMode == 'update' and $f->updateable() )
            )
            {
                $dataToSave[$f->getSource()] = $v->getWriteValue();
            }
        });
        return $dataToSave;
    }

    /**
     * [getDataToInsert description]
     * @return [type] [description]
     */
    function getDataToInsert()
    {
        return $this->getDataToSave('insert', false);
    }

    /**
     * [getDataToUpdate description]
     * @param  boolean $modifiedOnly [description]
     * @return [type]                [description]
     */
    function getDataToUpdate($modifiedOnly = false)
    {
        return $this->getDataToSave('update', $modifiedOnly);
    }

    /**
     * Insert record to database
     * @param  function  $callback 
     * @param  boolean $validate 
     * @param  boolean $recheck  
     * @return [type]            
     */
    function save( $callback = null, $validate = true, $recheck = true )
    {
        if ($validate) 
        {
            if (!$this->isValid($validate = true))
            {
                Helper::callback($callback, array(false, $this));
                return false;
            }
        }

        # check alive for decide to insert/update operation
        # update
        if ($this->isAlive($recheck))
        {
            $operation = Proxy::update(array(
                'table' => static::getTable(),
                'data' => $this->getDataToUpdate(),
                'condition' => static::mapToSource(array( static::getPrimary() => $this->getId() ))
            ));
        }
        # insert otherwise
        else 
        {
            $operation = Proxy::insert(array(
                'table' => static::getTable(),
                'data' => $this->getDataToInsert()
            ));
        }

        $success = $operation->isSuccess();

        if($success)
        {
            $this->reload($revert = true);
        }

        Helper::callback($callback, array($success, $this));

        return $success;
    }

    /**
     * Destroy a record from the database
     * @param  function $callback 
     * @return boolean           
     */
    function destroy( $callback = null )
    {
        $success = false;

        if($this->isAlive($recheck = true))
        {
            $condition = array(static::getPrimary() => $this->getId());

            $operation = Proxy::delete(array(
                'table' => static::getTable(),
                'condition' => static::mapToSource($condition),
            ));
            $success = (boolean)$operation->affectedRows; # should boolean instead

            # make the record as a ghost if destroyed
            if($success)
            {
                $this->makeAlive(false);
            }
        }

        Helper::callback($callback, array($success, $this));

        return $success;
    }

    /**
     * [relation description]
     * @return [type] [description]
     */
    protected function relation()
    {
        if(Collection::isNot($this->relation)) $this->relation = new Collection();
        return $this->relation;
    }

    /**
     * [getRelation description]
     * @return [type] [description]
     */
    function getRelation()
    {
        return $this->relation();
    }

    /**
     * Get the record/s of associated model
     * @param  /Roulette/Model  $association 
     * @param  fn  $callback    [description]
     * @param  boolean $reload      [description]
     * @return array               [description]
     */
    function associate( $association = null, $callback = null, $reload = true )
    {
        $relation = $this->relation();
        $association = $this->getAssociation($association);
        
        if ($association) 
        {
            return $association->associate($this, $reload);
            # Helper::callback($callback, array($association->getAssociated(), $association, $this));
        }
    }

    /**
     * Get the records of associated model
     * @param  /Roulette/Model  $association 
     * @param  boolean $reload      
     * @return array               
     */
    function lookup($association = null, $reload = false)
    {
        $assoc = $this->associate($association, $reload);
        if($assoc)
        {
            return $assoc->getResource();
        }
    }
