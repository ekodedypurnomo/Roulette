namespace Roulette\tunnel;

use Roulette\Base;

/**
 * Class Tunnel\Definer for defining the framework that being used.
 *
 * @package Roulette\Tunnel
 * @since Version 0.1.0
 * @author Eko Dedy Purnomo <eko.dedy.purnomo@gmail.com>
 */
abstract class Definer extends Base
{	
	/**
	 * Information of the used framework
	 * @var null
	 */
	static $frameworkInfo = null;

	/**
	 * Default value if framework not identified
	 * @return boolean
	 */
	static function check()
	{
		return false;
	}

	/**
	 * Get the information about the framework being used
	 * @return boolean
	 */
	static function info()
	{
		if(!static::$frameworkInfo)
		{
			static::check();
		}
		return static::$frameworkInfo;
	}

	/**
	 * @ignore
	 */
	static function getTunnel(){}
}
