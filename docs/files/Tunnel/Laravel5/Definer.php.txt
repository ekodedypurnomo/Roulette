namespace Roulette\Tunnel\Laravel5;

use Roulette\Tunnel\Definer as DefinerBase;

/**
 * Description Definer
 * 
 * @package Roulette\Tunnel\Laravel5
 * @since Version 0.1.0
 * @author Eko Dedy Purnomo <eko.dedy.purnomo@gmail.com>
 */
class Definer extends DefinerBase
{
	static $frameworkInfo = null;

	static function check()
	{
		$defined_functions = get_defined_functions()['user'];
        
        // check if laravel
        if(in_array('app', $defined_functions) and $app = app())
        {
            // require tunnel
            require_once(__DIR__.DIRECTORY_SEPARATOR."Tunnel.php");
            $tunnelClass = __NAMESPACE__."\Tunnel";
            $tunnel = new $tunnelClass('Illuminate\\Support\\Facades\\DB'); // if user use laravel so the DB is already defined

            $version = explode('.', $app::VERSION);
            
            static::$frameworkInfo = array(
                'framework' => 'Laravel',
                'version' => $app::VERSION,
                'versionInfo' => array(
                    'string'=> $app::VERSION,
                    'major'=> $version[0],
                    'minor'=> $version[1],
                    'patch'=> $version[2]
                ),
                'connection'=>'Illuminate\\Support\\Facades\\DB',
                'tunnel'=>$tunnel
            );
            return true;
        }

        return false;
	}
}
