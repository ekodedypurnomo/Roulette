namespace Roulette\Tunnel\Laravel5;

use Roulette\Tunnel\Tunnel as TunnelBase;
use Roulette\Operation;
use Roulette\Collection;

/**
 * The line to do trancasction to database via Laravel 5 application server
 * 
 * @package Roulette\Tunnel\Laravel5
 * @since Version 0.1.0
 * @author Eko Dedy Purnomo <eko.dedy.purnomo@gmail.com>
 */
class Tunnel extends TunnelBase
{
	protected $_loggingStatus = null;

	protected $_monitoringLog = false;

	protected $_initLogs = array(); 

	protected $_currentLogs = array(); 

	function __construct($connection = null)
	{
		$this->connection = $connection;
	}

	function getModel($model = null)
	{
		if(empty($model)) return;
		
		if(class_exists($model))
		{
			return $model;
		}
		else
		{
			try
			{
				require_once($model);
				return $model;
			} 
			catch (Exception $e) 
			{
				throw new Exception("Cannot load model '".$model."'", 1);
						
			}
		}
	}

	protected function startCaptureLog()
	{
		$conn = $this->getConnection();

		// backup old logging state
		$this->_loggingStatus = $conn::logging();

		$conn::enableQueryLog();

		$this->_monitoringLog = true;
		$this->_initLogs = $conn::getQueryLog();
	}

	protected function stopCaptureLog()
	{
		$conn = $this->getConnection();

		if($this->_loggingStatus === false) $conn::disableQueryLog();
		if($this->_loggingStatus === true) $conn::enableQueryLog();

		$this->_loggingStatus = null;
		$this->_monitoringLog = false;
		$this->_initLogs = array();
		$this->_currentLogs = array();
	}

	protected function getCapturedLog($applyBindings = true)
	{
		$conn = $this->getConnection();

		$this->_currentLogs = $conn::getQueryLog();

		$queryLog = null;

		// make sure logs is incremented after executing query
		if(count($this->_currentLogs) > count($this->_initLogs))
		{
			// then get the last of
			$queryLog = end($this->_currentLogs);

			// apply bindings
			if($applyBindings)
			{
				if($queryLog)
				{
					$q = $queryLog['query'];
			        foreach( $queryLog['bindings'] as $binding )
			        {
			        	// need more patch
			        	if(is_string($binding)) $binding = "'".$binding."'";
			        	if(is_null($binding)) $binding = 'NULL';

			            $q = preg_replace("#\?#", $binding, $q, 1);
			        }
			        $queryLog = $q;

			        // another trick
				    // # build a regular expression for each parameter
				    // $keys = array();
				    // foreach ($queryLog['bindings'] as $key => $value)
				    // {
				    //     $keys[] = is_string($key) ? '/:'.$key.'/' : '/[?]/';
				    // }
				    // $queryLog = preg_replace($keys, $queryLog['bindings'], $queryLog['query'], 1, $count);
				}
				else
				{
					$queryLog = null;
				}
			}
		}

		return $queryLog;
	}

	function operate(Operation $operation, callable $callback = null)
	{
		switch ($operation->action)
		{
			case 'query':
				$this->operateQuery($operation);
				break;

			case 'select':
				$this->operateSelect($operation);
				break;

			case 'exist':
				$this->operateExist($operation);
				break;

			case 'insert':
				$this->operateInsert($operation);
				break;

			case 'update':
				$this->operateUpdate($operation);
				break;

			case 'delete':
				$this->operateDelete($operation);
				break;

			case 'truncate':
				$this->operateTruncate($operation);
				break;
		}
		if(is_callable($callback))
		{
			call_user_func_array($callback, array($this, $operation));
		}
		return $this;
	}

	function operateQuery(Operation $operation)
	{
		$configs = Collection::from($operation->parameters)->setIfNot(array(
			'query' => null,
			'result_array' => false
		));

		$conn = $this->getConnection();
		$query = trim($configs->get('query'));

		// result
		if(!empty($query))
		{
			if(preg_match('/^select/i', $query))
			{
				$operation->result = $conn::select($query);
				$operation->affectedRows = 0;
			}
			elseif(preg_match('/^insert/i', $query))
			{
				$operation->result = $conn::insert($query);
				$operation->affectedRows = 0;
			}
			elseif(preg_match('/^update/i', $query))
			{
				$operation->result = $conn::update($query);
				$operation->affectedRows = $operation->result;
			}
			elseif(preg_match('/^delete/i', $query))
			{
				$operation->result = $conn::delete($query);
				$operation->affectedRows = $operation->result;
			}
			else{
				$operation->result = $conn::statement($query);
				$operation->affectedRows = 0;
			}
		}

		// affect to the operation package
		$operation->query = $query;
		$operation->rawQuery = $query;

		return $this;
	}

	function operateSelect(Operation $operation)
	{
		$configs = Collection::from($operation->parameters)->setIfNot(array(
			'table'			=> null,
			'fields'		=> '*',
			'condition'		=> null,
			'group' 		=> null,
			'having'		=> null,
			'skip' 			=> 0,
			'take' 			=> null,
			'sort'			=> null,
			'result_array' 	=> false
		));
		
		if(!$configs->has('table')) return $this;

		// table
		$conn = $this->getConnection();
		$builder = $conn::table($configs->get('table'));
		
		// fields
		$fields = $configs->get('fields');
		if(is_array($fields))
		{	
			foreach ($fields as $key => $value)
			{
				if(empty($value)) 
				{
					$value = $key;
					$fields[$key] = $key;
				}

				if($key != $value)
				{
					$fields[$key] = $value.' AS '.$key;
				}
			}
		}
		$builder->select($fields);
		
		// where
		if( $configs->has('condition') )
		{
			$condition = $configs->get('condition');	
			
			if( is_array($condition) )
			{
				foreach ($condition as $field => $value) 
				{	
					$builder->where($field, $value);
				}
			}
			else if( is_string($condition) )
			{
				$builder->where($condition, null);
			}
		}

		// group
		if( $configs->has('group') )
		{
			$builder->groupBy($configs->get('group'));
		
			// having
			if( $configs->has('having') )
			{
				$builder->havingRaw($configs->get('having'));
			}
		}

		// sort
		if( $configs->has('sort') )
		{
			$orders = $configs->get('sort');
			if( is_array($orders) )
			{
				foreach ( $orders as $field => $direction )
				{
	                if( in_array(strtolower($direction), array('asc','desc')) ){
	                    $builder->orderBy($field, $direction);
	                }else{
	                    $builder->orderby($field, 'asc');
	                }
	            }
			}else{
				$builder->orderBy($orders, 'asc');
			}
		}

		// take
		if( $configs->has('take') )
		{
			$builder->take($configs->get('take'));
		
			// skip
			if( $configs->has('skip') )
			{
				$builder->skip($configs->get('skip'));
			}
		}

		// activate query log
		$this->startCaptureLog();

		try
		{
			$result = $builder->get();
			
			// affect to the operation package
			$operation->success = true;
			$operation->records = $result;
			$operation->result = $result;
			$operation->affectedRows = 0; // there is no affected rows on select;
		} 
		catch (Exception $e) 
		{
			$operation->error = $e;	
			$operation->success = false;
		}

		// get last query log
		$operation->query = $this->getCapturedLog();
		$operation->rawQuery = $this->getCapturedLog(false);
		$this->stopCaptureLog();

		// some patch for compatibility with array
		if( $configs->get('result_array') === true )
		{
			foreach ($result as $i => $o)
			{
				$result[$i] = (array) $o;
			}

			$operation->result = $result;
		}

		return $this;
	}

	function operateExist( Operation $operation )
	{
		$configs = Collection::from($operation->parameters)->setIfNot(array(
			'table'		=> null,
			'condition'	=> array()
		));

		$conn = $this->getConnection();
		$builder = $conn::table($configs->get('table'));

		// activate query log
		$this->startCaptureLog();

		try 
		{
			$result = $builder->count("*");
		
			// affect to the operation package
			$operation->success = true;
			$operation->result = $result;
			$operation->affectedRows = 0; // there is no affected rows on insert;	
		} 
		catch (Exception $e) 
		{
			$operation->success = false;	
			$operation->error = $e;	
		}

		// get last query log
		$operation->query = $this->getCapturedLog();
		$operation->rawQuery = $this->getCapturedLog(false);
		$this->stopCaptureLog();

		return $this;
	}

	function operateInsert( Operation $operation )
	{
		$configs = Collection::from($operation->parameters)->setIfNot(array(
			'table'	=> null,
			'data'	=> null
		));

		$conn = $this->getConnection();
		$builder = $conn::table($configs->get('table'));

		// activate query log
		$this->startCaptureLog();

		try 
		{
			// result 
			$result = $builder->insert($configs->get('data'));
			
			// affect to the operation package
			$operation->success = true;
			$operation->result = $result;
			$operation->affectedRows = 0; // there is no affected rows on insert;
		} 
		catch (Exception $e) 
		{
			$operation->success = false;
			$operation->error = $e;
		}

		// get last query log
		$operation->query = $this->getCapturedLog();
		$operation->rawQuery = $this->getCapturedLog(false);
		$this->stopCaptureLog();

		return $this;
	}

	function operateUpdate( Operation $operation )
	{
		$configs = Collection::from($operation->parameters)->setIfNot(array(
			'table'		=> null,
			'data'		=> array(),
			'condition'	=> array()
		));

		$conn = $this->getConnection();
		$builder = $conn::table($configs->get('table'));

		// condition
		$builder->where($configs->get('condition'));

		// activate query log
		$this->startCaptureLog();

		try 
		{
			// result
			$result = $builder->update($configs->get('data'));

			// affect to the operation package
			$operation->success = true;
			$operation->result = $result;
			$operation->affectedRows = $result; // result is the affected rows;	
		} 
		catch (Exception $e) 
		{
			$operation->success = false;
			$operation->error = $e;	
		}
		
		// get last query log
		$operation->query = $this->getCapturedLog();
		$operation->rawQuery = $this->getCapturedLog(false);
		$this->stopCaptureLog();

		return $this;
	}

	function operateDelete( Operation $operation )
	{
		$configs = Collection::from($operation->parameters)->setIfNot(array(
			'table'		=> null,
			'condition' => array()
		));

		// table
		$conn = $this->getConnection();
		$builder = $conn::table($configs->get('table'));

		// condition
		$builder->where($configs->get('condition'));

		// activate query log
		$this->startCaptureLog();

		try 
		{
			// result
			$result = $builder->delete();
			
			// affect to the operation package
			$operation->success = true;
			$operation->result = $result;
			$operation->affectedRows = $result; // result is number of affected rows;	
		} 
		catch (Exception $e) 
		{
			$operation->success = false;
			$operation->error = $e;		
		}

		// get last query log
		$operation->query = $this->getCapturedLog();
		$operation->rawQuery = $this->getCapturedLog(false);
		$this->stopCaptureLog();

		return $this;
	}

	function operateTruncate( Operation $operation )
	{
		$configs = Collection::from($operation);

		$configs->setIfNot(array(
			'table'		=> null
		));

		// table
		$conn = $this->getConnection();
		$builder = $conn::table($configs->get('table'));

		try 
		{
			// result
			$result = $builder->delete();

			// affect to the operation package
			$operation->success = true;
			$operation->result = $result;
			$operation->affectedRows = $result; // result is number of affected rows;	
		}
		catch (Exception $e)
		{
			$operation->success = false;	
			$operation->error = $e;	
		}

		// get last query log
		$operation->query = $this->getCapturedLog();
		$operation->rawQuery = $this->getCapturedLog(false);
		$this->stopCaptureLog();

		return $this;
	}

}
