namespace Roulette\tunnel;

use Roulette\Base;
use Roulette\Operation;

/**
 * Abstract class for Tunnel.
 *
 * @package Roulette\Tunnel
 * @since Version 0.1.0
 * @author Eko Dedy Purnomo <eko.dedy.purnomo@gmail.com>
 */
abstract class Tunnel extends Base
{

	/**
	 * Config connection to the framework
	 * @var null
	 */
	protected $connection = null;

	/**
	 * @ignore
	 */
	function __construct($connection = null)
	{
		$this->connection = $connection;
	}

	/**
	 * Get the connection configuration
	 * @return array
	 */
	function getConnection()
	{
		return $this->connection;
	}

	/**
	 * Abstract function of operate
	 * @param  Operation $operation 
	 * @param  callable  $callback  
	 */
	abstract function operate(Operation $operation, callable $callback);

}
