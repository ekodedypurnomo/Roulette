namespace Roulette\validation;

use Roulette\Base;
use Roulette\validation\validator\Validator;
use Roulette\validation\Validation\Validator\String;
use Roulette\validation\Validation\Validator\Boolean;
use Roulette\validation\Validation\Validator\Numeric;
use Roulette\validation\Validation\Validator\Float;
use Roulette\validation\Validation\Validator\Double;
use Roulette\validation\Validation\Validator\Integer;
use Roulette\validation\Validation\Validator\Above;
use Roulette\validation\Validation\Validator\Below;
use Roulette\validation\Validation\Validator\Exlusion;
use Roulette\validation\Validation\Validator\Inclusion;
use Roulette\validation\Validation\Validator\Maxlength;
use Roulette\validation\Validation\Validator\Maxvalue;
use Roulette\validation\Validation\Validator\Minlength;
use Roulette\validation\Validation\Validator\MinValue;
use Roulette\validation\Validation\Validator\Nullable;
use Roulette\validation\Validation\Validator\Format;
use Roulette\validation\Validation\Validator\Custom;

use Roulette\util\Configurable;

/**
 * Validation is the appearance of the message or perform other functions after action
 * 
 * @package \Roulette
 * @since Version 0.1.0
 * @author Eko Dedy Purnomo <eko.dedy.purnomo@gmail.com>
 */
class Validation extends Base
{
    // use util\Observable;
    use Configurable;
    
    /**
     * Default value for validators
     * @var array
     */
    protected $validators = array();
    
    /**
     * Default value for validator message
     * @var array
     */
    protected $validatorsMessage = array();

    /**
     *@ignore
     */
    function __construct($config = null)
    {        
        if( is_callable($config) )
        {
            $config = array('validators'=>array(
                'custom'=>$config
            ));
        }
        
        $this->configure($config);

        // backup affected from config
        $validators = is_array($this->validators) ? $this->validators : array();
        // before goes deep, purge any artifact
        $this->validators = array();
        // now append each
        foreach ($validators as $validator => $rule)
        {
            if(Validator::is($rule))
            {
                $this->addValidator($rule);
            }
            else
            {
                $this->addValidator($validator, $rule);
            }
        }
        
        return $this;
    }

    /**
     * Take specified validators
     * @return Array
     */
    function getValidators()
    {
        return $this->validators;
    }

    /**
     * Add new validator
     * 
     * @param Array  $validator
     * @param Array  $rule
     * @param String $message
     */
    function addValidator($validator = null, $rule = null, $message = null)
    {
        if (Validator::is($validator))
        {
            $this->validators[] = $validator;
            return $this;
        }

        // buildin validator
        $builtinValidators = array(
            'nullable'  => 'Nullable',
            'format'    => 'Format',
            'inclusion' => 'Inclusion',
            'exclusion' => 'Exclusion',
            'minvalue'  => 'Minvalue',
            'maxvalue'  => 'Maxvalue',
            'above'     => 'Above',
            'below'     => 'Below',
            'minlength' => 'Minlength',
            'maxlength' => 'Maxlength',
            'custom'    => 'Custom'
        );

        $validator = strtolower($validator);
        
        if(array_key_exists($validator, $builtinValidators))
        {
            $validatorClass = "Roulette\\validation\\validator\\".$builtinValidators[$validator];

            if(array_key_exists($validator, $this->validatorsMessage)) $message = $this->validatorsMessage[$validator];

            $this->validators[] = new $validatorClass($rule, $message);
        }
        return $this;
    }

    /**
     * Reset validors that has been entered
     * @return array
     */
    function reset()
    {
        $this->validators = array();
        return $this;
    }

    /**
     * Take specified validator message
     * 
     *     Example :
     *     $message = array(
     *             'validators'=>array(),
     *             'validatorMessage'=>array('null'=>'dont null please',
     *                                       'sing'=>'lalalala'
     *                                       )
     *     );
     *
     *     $msg = \Roulette\validation\Validation::getValidatorMessage('null') == 'dont null please';
     *     
     * @return Array
     */
    function getValidatorsMessage()
    {
        $args = func_get_args();
        if (empty($args))
        {
            return $this->validatorsMessage;
        } 
        else
        {
            $key = $args[0];
            if (array_key_exists($key, $this->validatorsMessage))
            {
                return $this->validatorsMessage[$key];
            }
        }
    }

    /**
     * Matching validator is the same
     * 
     * @param  Array $value
     * @return Array
     */
    function validate($value = null)
    {
        $isValid = true;
        $validationMessages = array();

        foreach ( $this->validators as $validator)
        {
            if( $validator->test($value) !== true )
            {
                $isValid = false;
                $validationMessages[] = $validator->getMessage(array(
                    'value'=>$value
                ));
            }
        }

        return $validationMessages;
    }

}
