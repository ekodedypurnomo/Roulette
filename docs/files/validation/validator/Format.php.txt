namespace Roulette\validation\validator;

use Roulette\validation\validator\Validator;

/**
 * Built-in validators.
 * Validate value with regex as a tester.
 * 
 * Example:
 * ```php
 * 	$validator = new Roulette\validation\validator\Format(
 * 		'/foo/', '"{value}" does not pass the test.'
 * 	);
 * 	$validator->test('bar'); // return `false`
 * 	$validator->getMessage(); // return `"bar" does not pass the test`
 * ```
 *
 * @package Roulette\Validator
 * @since Version 0.1.0
 * @author Eko Dedy Purnomo <eko.dedy.purnomo@gmail.com>
 * @see  Validator
 */
class Format extends Validator
{
	/**
	 * Default value for format message
	 * @var string
	 */
	protected $message = 'invalid format';

	/**
	 * Execute the proses validation
	 * 
	 * @param  string $value variable to be validated
	 * @return boolean true if the variable is valid
	 */
    function test($value = null)
    {
    	return is_string($value) && preg_match($this->rule, $value);
    }
}
