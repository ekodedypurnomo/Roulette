namespace Roulette;

use Roulette\Base;
use Roulette\Tunnel\tunnels;
use Roulette\Tunnel\Tunnel;
use Roulette\Tunnel\Definer;

/**
 * Check the type of the framework used by the server application.
 *
 * @package \Roulette
 * @since Version 0.1.0
 * @author Eko Dedy Purnomo <eko.dedy.purnomo@gmail.com>
 */
class FrameworkChecker extends Base
{	
	/**
	 * [$frameworks description]
	 * @var null
	 */
	protected static $frameworks = null;
	
	/**
	 * [$configLoaded description]
	 * @var boolean
	 */
	protected static $configLoaded = false;

	/**
	 * [getFrameworks description]
	 * @return [type] [description]
	 */
	static function getFrameworks()
	{
		if(!static::$configLoaded)
		{
			static::$frameworks = require_once(__DIR__.'\Tunnel\tunnels.php');
			static::$configLoaded = true;
			if(!is_array(static::$frameworks)) static::$frameworks = array();
		}
		return static::$frameworks;
	}

	/**
	 * [addFramework description]
	 * @param [type]   $frameworkName [description]
	 * @param callable $validator     [description]
	 */
	static function addFramework($frameworkName, callable $validator)
	{

	}

	/**
	 * [hasFramework description]
	 * @param  [type]  $frameworkName [description]
	 * @return boolean                [description]
	 */
	static function hasFramework($frameworkName = null)
	{
		$has = false;
		if(empty($frameworkName)) return false;

		foreach (static::getFrameworks() as $key => $value) 
		{
			$has = preg_match('/'.strtolower($frameworkName).'/', strtolower($key));
			if($has)break;
		}
		return $has;
	}

	/**
	 * [getFrameworkInfo description]
	 * @return [type] [description]
	 */
	static function getFrameworkInfo()
	{
		foreach (static::getFrameworks() as $key => $definer)
		{
			if(!class_exists($definer) and !empty($definer))
			{
				include_once(dirname(__DIR__)."\\".$definer.".php");
			}

			if(class_exists($definer) and is_callable("$definer::check"))
			{
				$valid = $definer::check();
				if($valid and is_callable("$definer::info"))
				{
					return $definer::info();
				}
				// break manualy if info return is uncallable
				break;
			}
		}
	}
}
