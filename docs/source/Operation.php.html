namespace Roulette;

use Roulette\Base;
use Roulette\Tunnel\Tunnel;

/**
 * Standardize database request operation
 *
 * @package \Roulette
 * @since Version 0.1.0
 * @author Eko Dedy Purnomo <eko.dedy.purnomo@gmail.com>
 */
class Operation extends Base
{
    // use util\Observable;
    use Util\Configurable;

    /**
     * [$tunnel description]
     * @var null
     */
    protected $tunnel = null;
    
    /**
     * actions to be taken by the application
     * @var string
     */
    public $action = 'query'; //select|insert|update|delete|query on proxy

    /**
     * create a function with the number of parameters that can change
     * @var null
     */
    public $parameters = null;

    /**
     * notification that an action has been successful
     * @var null
     */
    public $success = null;

    /**
     * notification that an action has occurred error
     * @var null
     */
    public $error = null;

    /**
     * retrieve data on a model
     * @var null
     */
    public $records = null;

    /**
     * Last query executed
     * @var null
     */
    public $query = null;

    /**
     * Original last query from framework 
     * @var null
     */
    public $rawQuery = null;

    /**
     * Rows that affected by last operation
     * @var null
     */
    public $affectedRows = null;

    /**
     * The result fo the operation
     * @var null
     */
    public $result = null;

    /**
     * process with database
     * @var boolean
     */
    public $executed = false;

    /**
     * [$executeStart description]
     * @var null
     */
    public $executeStart = null;

    /**
     * [$executeFinish description]
     * @var null
     */
    public $executeFinish = null;

    /**
     * The reququired time to execute the operation
     * @var null
     */
    public $executeTime = null;

    /**
     * @ignore
     */
    function __construct( Tunnel $tunnel, $config = array() )
    {
        $this->configure($config);
        $this->tunnel = $tunnel;
        return $this;
    }
    
    /**
     * the process of processing data to be sent to DB
     * @return [type] [description]
     */
    function getTunnel()
    {
        return $this->tunnel;
    }
    
    /**
     * send or retrieve data
     * @return Array
     */
    function getRecord()
    {
        return ( is_array($this->records) and !empty($this->records) ) ? $this->records[0] : null;
    }

    /**
     * Get the records from the operation
     * @return Array
     */
    function getRecords()
    {
        if(!is_array($this->records)) $this->records = array();
        return $this->records;
    }

    /**
     * check whether the data error
     * @return [type] [description]
     */
    function getError()
    {
        return $this->error;
    }

    /**
     * Check whether the operation is a success
     * @return boolean [description]
     */
    function isSuccess()
    {
        return $this->success;
    }

    /**
     * Check Whether the operation is ececuted or not
     * @return boolean [description]
     */
    function isExecuted()
    {
        return (bool)$this->executed;
    }

    /**
     * function performs a process with database
     * @return [type] [description]
     */
    function execute()
    {
        // initialized
        $this->executeStart = microtime(true);
        $this->executeFinish = null;
        $this->executeTime = null;
        $this->records = null;
        $this->result = null;
        $this->success = false;
        $this->query = null;
        $this->rawQuery = null;
        $this->affectedRows = null;
        
        // executing
        if($tunnel = $this->getTunnel())
        {
            $tunnel->operate($this, function($t, $o)
            {
                // apply execute time data
                $o->executed = true;
                $o->executeFinish = microtime(true);
                $o->executeTime = round($o->executeFinish - $o->executeStart, 3);
            });
        }
        
        return $this;
    }

    /**
     * function sends that an action has been successful
     * 
     * @param  callable|null $callback [description]
     * @return [type]                  [description]
     */
    function success(callable $callback = null)
    {
        if($this->isSuccess())
            call_user_func_array($callback, array($this));

        return $this;
    }

    /**
     * function sends that an action has failed
     * 
     * @param  callable|null $callback [description]
     * @return [type]                  [description]
     */
    function failure(callable $callback = null)
    {
        if(!$this->isSuccess()) 
            call_user_func_array($callback, array($this));

        return $this;
    }

    /**
     * function displays a notification after the previous effect
     * 
     * @param  callable|null $callback [description]
     * @return function                [description]
     */
    function callback(callable $callback = null)
    {
        call_user_func_array($callback, array($this, $this->isSuccess()));

        return $this;
    }

}
