/**
 * Roulette\association\Association package contain any class which support relation between models
 */
namespace Roulette\association;

use Roulette\Base;
use Roulette\field\Field;
use Roulette\Proxy;
use Roulette\Collection;
use Roulette\Iterator;
use Roulette\association\Value as AssociationValue;

use Roulette\util\Configurable;

/**
 * Assosiation was a description of a relationship between a model one with the other models
 * which in this function there-many relationship model or one model
 * 
 * @package \Roulette
 * @since Version 0.1.0
 * @author Eko Dedy Purnomo <eko.dedy.purnomo@gmail.com>
 */
class Association extends Base
{   
    use Configurable;

    /**
     * name of the Association
     * @var String
     */
    protected $name = null;
    
    /**
     * Type of the Association 
     * only hasMany | hasOne
     * @var string
     */
    protected $type = 'hasMany';
    
    /**
     * Foreign field of the associated model
     * @var null
     */
    protected $field = null;

    /**
     * the Name of the model that being associated with
     * @var String
     */
    protected $model = null;

    /**
     * [__construct description]
     * 
     * @param [type] $config [description]
     */
    function __construct($config = null)
    {
        if(is_string($config)) $config = array('name'=>$config);
        if(is_object($config)) $config = (array) $config; 
        if(!is_array($config)) $config = array($config);

        $configs = Collection::create($config);

        $configs->set(array(
            'type' => Iterator::enum($configs->get('type'), array('hasMany','hasOne','belongsTo'), 'hasMany')
        ));
        $configs->setIfNot(array(
            'name' => $configs->get('model')
        ));

        $this->configure($configs->getAll());

        return $this;
    }

    /**
     * get the Name fo the Association
     * @return String
     */
    function getName()
    {
        return $this->name;
    }

    /**
     * get the Type of the Association
     * @return String
     */
    function getType()
    {
        return $this->type;
    }

    /**
     * get the name of the model that being Associated with
     * @return /Roulette/MOdel 
     */
    function getModel()
    {
        return Proxy::getModel($this->model);
    }

    /**
     * get the foreign field of the assocaited model
     * @return String [description]
     */
    function getField()
    {
        return $this->field;
    }

    /**
     * Get record/s from the model that being associated with
     * 
     * @param  array  $record retrieve data on a model
     * @param  boolean $reload choice whether or not to refresh
     * @return array [description]
     */
    function associate( $record, $reload = true )
    {
        $name = $this->getName();
        $model = $this->getModel();
        $field = $this->getField();
        $type = $this->getType();

        $value = $record->get($field, false); // get resource
        $assocValue = $record->getRelation()->get($name);

        if(!$assocValue)
        {
            $assocValue = new AssociationValue($this, $record);
        }

        if($type == 'hasOne' or $type == "belongsTo")
        {
            $assocValue->associated = true;
            $assocValue->resource = $model::load($value);
        }
        else if($type == 'hasMany')
        {
            $assocValue->associated = true;
            $assocValue->resource = [];
            
            $assocValue->resource = $model::find($field, $value);    
        }

        return $assocValue;
    }

}
