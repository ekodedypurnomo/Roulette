namespace Roulette\validation\validator;

use Roulette\validation\validator\Validator;

/**
 * SubClass for validator, will be show message "must be exlude from {validator}"
 * 
 * @package \Roulette\Validator
 * @since Version 0.1.0
 * @author Eko Dedy Purnomo <eko.dedy.purnomo@gmail.com>
 */
class Exclusion extends Validator
{
    /**
     * Default validator message for Exclusion
     * @var string
     */
	protected $message = 'must be exclude from: <b>{validator}</b>';

    /**
     * Default config for Exclusion Validator
     * @param Array  $validator
     * @param String $message
     */
	function __construct($validator = null, $message = null)
	{
		parent::__construct($validator, $message);
		if(! is_array($this->rule) ) $this->rule = array();
	}

    /**
     * Execute the validation process
     * @param  string $value variable to be validated
     * @return boolean true if the variable is valid 
     */
    function test($value = null)
    {
        return !in_array($value, $this->rule);
    }

    /**
     * Take Specified validator
     * @return String
     */
    function getRuleString()
    {
        if(!is_array($this->rule)) $this->rule = array();
        return implode(',',$this->rule);
    }
}
